class Solution:
    def sortColors(self, nums):
        l=0
        h=len(nums)-1
        mid=0
        while mid<=h:
            if nums[mid]==0:
                nums[l], nums[mid] = nums[mid], nums[l]
                mid+=1
                l+=1
            elif nums[mid]==1:
                nums[mid]=1
                mid+=1
            else:
                nums[mid], nums[h] = nums[h], nums[mid]
                h-=1
        return nums
